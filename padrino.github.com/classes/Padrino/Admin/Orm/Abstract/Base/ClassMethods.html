            def column_store(*args)
              path   = File.join(*args)
              path   = Dir[path + ".{jml,jaml}"].first.to_s if path !~ /(\.jml|\.jaml)$/
              config = YAML.load_file(path)
              Padrino::Admin::ColumnStore.new(self, config)
            end
